220 /*
221 A dynamic array of Plain Old Data. Doesn't support constructors, etc.
222 Has a small initial memory pool, so that low or no usage will not
223 cause a call to new/delete
224 */
225 template <class T, int INITIAL_SIZE>
226 class DynArray
227 {
228 public:
229 DynArray() : _mem(_pool),
230 _allocated(INITIAL_SIZE),
231 _size(0)
232 {
233 }
234
235 ~DynArray()
236 {
237 if (_mem != _pool)
238 {
239 delete[] _mem;
240 }
241 }
242
243 void Clear()
244 {
245 _size = 0;
246 }
247
248 void Push(T t)
249 {
250 TIXMLASSERT(_size < INT_MAX);
251 EnsureCapacity(_size + 1);
252 _mem[_size] = t;
253 ++_size;
254 }
255
256 T *PushArr(int count)
257 {
258 TIXMLASSERT(count >= 0);
259 TIXMLASSERT(_size <= INT_MAX - count);
260 EnsureCapacity(_size + count);
261 T *ret = &_mem[_size];
262 _size += count;
263 return ret;
264 }
265
266 T Pop()
267 {
268 TIXMLASSERT(_size > 0);
269 --_size;
270 return _mem[_size];
271 }
272
273 void PopArr(int count)
274 {
275 TIXMLASSERT(_size >= count);
276 _size -= count;
277 }
278
279 bool Empty() const
280 {
281 return _size == 0;
282 }
283
284 T &operator[](int i)
285 {
286 TIXMLASSERT(i >= 0 && i < _size);
287 return _mem[i];
288 }
289
290 const T &operator[](int i) const
291 {
292 TIXMLASSERT(i >= 0 && i < _size);
293 return _mem[i];
294 }
295
296 const T &PeekTop() const
297 {
298 TIXMLASSERT(_size > 0);
299 return _mem[_size - 1];
300 }
301
302 int Size() const
303 {
304 TIXMLASSERT(_size >= 0);
305 return _size;
306 }
307
308 int Capacity() const
309 {
310 TIXMLASSERT(_allocated >= INITIAL_SIZE);
311 return _allocated;
312 }
313
314 void SwapRemove(int i)
315 {
316 TIXMLASSERT(i >= 0 && i < _size);
317 TIXMLASSERT(_size > 0);
318 _mem[i] = _mem[_size - 1];
319 --_size;
320 }
321
322 const T *Mem() const
323 {
324 TIXMLASSERT(_mem);
325 return _mem;
326 }
327
328 T *Mem()
329 {
330 TIXMLASSERT(_mem);
331 return _mem;
332 }
333
334 private:
335 DynArray(const DynArray &); // not supported
336 void operator=(const DynArray &); // not supported
337
338 void EnsureCapacity(int cap)
339 {
340 TIXMLASSERT(cap > 0);
341 if (cap > _allocated)
342 {
343 TIXMLASSERT(cap <= INT_MAX / 2);
344 const int newAllocated = cap * 2;
345 T *newMem = new T[newAllocated];
346 TIXMLASSERT(newAllocated >= _size);
347 memcpy(newMem, _mem, sizeof(T) * _size); // warning: not using constructors, only works for PODs
348 if (_mem != _pool)
349 {
350 delete[] _mem;
351 }
352 _mem = newMem;
353 _allocated = newAllocated;
354 }
355 }
356
357 T *_mem;
358 T _pool[INITIAL_SIZE];
359 int _allocated; // objects allocated
360 int _size; // number objects in use
361 };

379 /*
380 Template child class to create pools of the correct type.
381 */
382 template <int ITEM_SIZE>
383 class MemPoolT : public MemPool
384 {
385 public:
386 MemPoolT() : _blockPtrs(), _root(0), _currentAllocs(0), _nAllocs(0), _maxAllocs(0), _nUntracked(0) {}
387 ~MemPoolT()
388 {
389 MemPoolT<ITEM_SIZE>::Clear();
390 }
391
392 void Clear()
393 {
394 // Delete the blocks.
395 while (!_blockPtrs.Empty())
396 {
397 Block *lastBlock = _blockPtrs.Pop();
398 delete lastBlock;
399 }
400 _root = 0;
401 _currentAllocs = 0;
402 _nAllocs = 0;
403 _maxAllocs = 0;
404 _nUntracked = 0;
405 }
406
407 virtual int ItemSize() const
408 {
409 return ITEM_SIZE;
410 }
411 int CurrentAllocs() const
412 {
413 return _currentAllocs;
414 }
415
416 virtual void *Alloc()
417 {
418 if (!_root)
419 {
420 // Need a new block.
421 Block *block = new Block;
422 _blockPtrs.Push(block);
423
424 Item *blockItems = block->items;
425 for (int i = 0; i < ITEMS_PER_BLOCK - 1; ++i)
426 {
427 blockItems[i].next = &(blockItems[i + 1]);
428 }
429 blockItems[ITEMS_PER_BLOCK - 1].next = 0;
430 _root = blockItems;
431 }
432 Item *const result = _root;
433 TIXMLASSERT(result != 0);
434 _root = _root->next;
435
436 ++_currentAllocs;
437 if (_currentAllocs > _maxAllocs)
438 {
439 _maxAllocs = _currentAllocs;
440 }
441 ++_nAllocs;
442 ++_nUntracked;
443 return result;
444 }
445
446 virtual void Free(void *mem)
447 {
448 if (!mem)
449 {
450 return;
451 }
452 --_currentAllocs;
453 Item *item = static_cast<Item *>(mem);
454#ifdef TINYXML2_DEBUG
455 memset(item, 0xfe, sizeof(*item));
456#endif
457 item->next = _root;
458 _root = item;
459 }
460 void Trace(const char *name)
461 {
462 printf("Mempool %s watermark=%d [%dk] current=%d size=%d nAlloc=%d blocks=%d\n",
463 name, _maxAllocs, _maxAllocs * ITEM_SIZE / 1024, _currentAllocs,
464 ITEM_SIZE, _nAllocs, _blockPtrs.Size());
465 }
466
467 void SetTracked()
468 {
469 --_nUntracked;
470 }
471
472 int Untracked() const
473 {
474 return _nUntracked;
475 }

615 static const char *SkipWhiteSpace(const char *p, int *curLineNumPtr)
616 {
617 TIXMLASSERT(p);
618
619 while (IsWhiteSpace(*p))
620 {
621 if (curLineNumPtr && *p == '\n')
622 {
623 ++(*curLineNumPtr);
624 }
625 ++p;
626 }
627 TIXMLASSERT(p);
628 return p;
629 }

631 {
632 return const_cast<char *>(SkipWhiteSpace(const_cast<const char *>(p), curLineNumPtr));
633 }

636 // correct, but simple, and usually works.
637 static bool IsWhiteSpace(char p)
638 {
639 return !IsUTF8Continuation(p) && isspace(static_cast<unsigned char>(p));
640 }

642 inline static bool IsNameStartChar(unsigned char ch)
643 {
644 if (ch >= 128)
645 {
646 // This is a heuristic guess in attempt to not implement Unicode-aware isalpha()
647 return true;
648 }
649 if (isalpha(ch))
650 {
651 return true;
652 }
653 return ch == ':' || ch == '_';
654 }
655
656 inline static bool IsNameChar(unsigned char ch)
657 {
658 return IsNameStartChar(ch) || isdigit(ch) || ch == '.' || ch == '-';
659 }
660
661 inline static bool IsPrefixHex(const char *p)
662 {
663 p = SkipWhiteSpace(p, 0);
664 return p && *p == '0' && (*(p + 1) == 'x' || *(p + 1) == 'X');
665 }

674 TIXMLASSERT(q);
675 TIXMLASSERT(nChar >= 0);
