src/app/main/store/reducers.ts

initialState
Type : MainStateInterface
Default value : { message: '', isLoadingMessage: false, messages: [], }
reducer
Default value : createReducer( initialState, on( sendMessageAction, (state, action): MainStateInterface => ({ ...state, isLoadingMessage: true, messages: [...state.messages, { send: action.message, receive: '' }], }) ), on( sendMessageSuccessAction, (state, action): MainStateInterface => ({ ...state, isLoadingMessage: false, message: action.response.message, messages: state.messages.map((message) => { if (message.receive === '') { return { send: message.send, receive: action.response.message }; } else return message; }), }) ), on( sendMessageFailureAction, (state): MainStateInterface => ({ ...state, isLoadingMessage: false, messages: state.messages.slice(0, -1), }) ) )
